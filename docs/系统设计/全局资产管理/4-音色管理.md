# 音色管理设计文档

## 1. 概述

音色是用户级别的独立资产，存储于资产中心，可被多个角色复用。

| 特性 | 说明 |
|------|------|
| 归属级别 | 用户级，非项目级 |
| 独立性 | 完全独立，不依赖角色 |
| 创建方式 | AI 设计 / 上传音频 |
| 复用机制 | 一个音色可被多个角色引用 |

## 3. 音色类型

### 3.1 两种音色类型

| voiceType | 说明 | 来源 |
|-----------|------|------|
| qwen-designed | AI 设计音色 | 通义千问 TTS 生成 |
| custom | 自定义音色 | 用户上传音频文件 |

### 3.2 类型差异

```
qwen-designed（AI 设计）
├── qwenVoiceId: "xxx"          ← qwen-tts 返回的音色ID
├── voicePrompt: "温柔的女声"    ← 设计时使用的提示词
└── customVoiceUrl: null

custom（自定义上传）
├── qwenVoiceId: null
├── customVoiceUrl: "https://..." ← 用户上传的音频URL
└── customVoiceMediaId: "xxx"    ← 媒体对象ID
```

## 4. 数据关系

```
User
│
├── GlobalAssetFolder (folderId 可选)
│
└── GlobalVoice（音色）
│
├── MediaObject (customVoiceMedia, 可选)
│
└── 被引用
    │
    ├── GlobalCharacter.globalVoiceId ──┐
    │                                   │ 多对多引用
    └── NovelPromotionCharacter.globalVoiceId
```

**关系特点：**
- 音色独立存在，不依赖角色
- 多个角色可以引用同一个音色
- 删除音色时，角色音色字段会被清空，但角色不会被删除

## 7. Service 层实现

### 7.1 VoiceService

```java
package io.novel2video.agent.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import io.novel2video.agent.entity.*;
import io.novel2video.agent.mapper.*;
import io.novel2video.agent.ai.QwenTtsService;
import io.novel2video.agent.storage.MinioService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.multipart.MultipartFile;

import java.util.*;

@Slf4j
@Service
@RequiredArgsConstructor
public class VoiceService {

    private final GlobalVoiceMapper voiceMapper;
    private final GlobalCharacterMapper characterMapper;
    private final QwenTtsService qwenTtsService;
    private final MinioService minioService;

    /**
     * 获取音色列表
     */
    public List<GlobalVoice> listVoices(String userId, String folderId) {
        LambdaQueryWrapper<GlobalVoice> wrapper = new LambdaQueryWrapper<GlobalVoice>()
                .eq(GlobalVoice::getUserId, userId);
        
        if ("null".equals(folderId)) {
            wrapper.isNull(GlobalVoice::getFolderId);
        } else if (folderId != null) {
            wrapper.eq(GlobalVoice::getFolderId, folderId);
        }
        
        wrapper.orderByDesc(GlobalVoice::getCreatedAt);
        return voiceMapper.selectList(wrapper);
    }

    /**
     * 获取单个音色
     */
    public GlobalVoice getVoice(String voiceId) {
        return voiceMapper.selectByVoiceId(voiceId);
    }

    /**
     * 创建音色
     */
    public GlobalVoice createVoice(String userId, CreateVoiceRequest request) {
        GlobalVoice voice = new GlobalVoice()
                .setVoiceId(UUID.randomUUID().toString())
                .setUserId(userId)
                .setName(request.getName())
                .setDescription(request.getDescription())
                .setFolderId(request.getFolderId())
                .setQwenVoiceId(request.getQwenVoiceId())
                .setVoiceType(request.getVoiceType() != null ? request.getVoiceType() : "qwen-designed")
                .setCustomVoiceUrl(request.getCustomVoiceUrl())
                .setVoicePrompt(request.getVoicePrompt())
                .setGender(request.getGender())
                .setLanguage(request.getLanguage() != null ? request.getLanguage() : "zh");
        
        voiceMapper.insert(voice);
        log.info("Created voice: {}", voice.getVoiceId());
        return voice;
    }

    /**
     * 上传音频创建音色
     */
    @Transactional
    public GlobalVoice uploadVoice(String userId, MultipartFile file, 
            String name, String folderId, String description, String gender, String language) {
        
        // 1. 上传音频到 MinIO
        String url = minioService.upload(file, "voices/" + userId + "/");
        
        // 2. 创建音色记录
        GlobalVoice voice = new GlobalVoice()
                .setVoiceId(UUID.randomUUID().toString())
                .setUserId(userId)
                .setName(name)
                .setDescription(description)
                .setFolderId(folderId)
                .setVoiceType("custom")
                .setCustomVoiceUrl(url)
                .setGender(gender)
                .setLanguage(language != null ? language : "zh");
        
        voiceMapper.insert(voice);
        log.info("Created voice from upload: {}", voice.getVoiceId());
        return voice;
    }

    /**
     * 更新音色
     */
    public GlobalVoice updateVoice(String voiceId, UpdateVoiceRequest request) {
        GlobalVoice voice = voiceMapper.selectByVoiceId(voiceId);
        if (voice == null) {
            throw new RuntimeException("Voice not found: " + voiceId);
        }
        
        if (request.getName() != null) {
            voice.setName(request.getName());
        }
        if (request.getDescription() != null) {
            voice.setDescription(request.getDescription());
        }
        if (request.getFolderId() != null) {
            voice.setFolderId("null".equals(request.getFolderId()) ? null : request.getFolderId());
        }
        if (request.getQwenVoiceId() != null) {
            voice.setQwenVoiceId(request.getQwenVoiceId());
        }
        if (request.getVoiceType() != null) {
            voice.setVoiceType(request.getVoiceType());
        }
        if (request.getCustomVoiceUrl() != null) {
            voice.setCustomVoiceUrl(request.getCustomVoiceUrl());
        }
        if (request.getGender() != null) {
            voice.setGender(request.getGender());
        }
        if (request.getLanguage() != null) {
            voice.setLanguage(request.getLanguage());
        }
        
        voiceMapper.updateById(voice);
        log.info("Updated voice: {}", voiceId);
        return voice;
    }

    /**
     * 删除音色
     */
    @Transactional
    public void deleteVoice(String voiceId, String userId) {
        GlobalVoice voice = voiceMapper.selectByVoiceId(voiceId);
        if (voice == null) {
            return;
        }
        
        // 1. 清空引用此音色的角色的 globalVoiceId
        characterMapper.clearGlobalVoiceId(voiceId, userId);
        
        // 2. 如果是百炼音色，检查是否需要清理
        if ("qwen-designed".equals(voice.getVoiceType()) && voice.getQwenVoiceId() != null) {
            // 检查是否有其他音色引用同一个百炼音色
            boolean hasOtherRef = voiceMapper.existsByQwenVoiceIdAndNotVoiceId(
                    voice.getQwenVoiceId(), voiceId, userId);
            if (!hasOtherRef) {
                // 调用百炼 API 清理音色
                qwenTtsService.deleteVoice(voice.getQwenVoiceId());
                log.info("Cleaned up qwen voice: {}", voice.getQwenVoiceId());
            }
        }
        
        // 3. 删除音色记录
        voiceMapper.deleteByVoiceId(voiceId, userId);
        log.info("Deleted voice: {}", voiceId);
    }

    /**
     * AI 设计音色
     */
    public VoiceDesignResponse designVoice(VoiceDesignRequest request) {
        // 调用 qwen-tts-vd API
        return qwenTtsService.designVoice(request.getSpeaker(), request.getPrompt());
    }

    /**
     * 获取音色选择器列表
     */
    public List<GlobalVoice> getVoicePicker(String userId) {
        LambdaQueryWrapper<GlobalVoice> wrapper = new LambdaQueryWrapper<GlobalVoice>()
                .eq(GlobalVoice::getUserId, userId)
                .orderByDesc(GlobalVoice::getCreatedAt);
        return voiceMapper.selectList(wrapper);
    }
}
```

## 9. AI 音色设计

### 9.1 QwenTtsService

```java
package io.novel2video.agent.ai;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.web.reactive.function.client.WebClient;

@Slf4j
@Service
@RequiredArgsConstructor
public class QwenTtsService {

    private final WebClient qwenTtsClient;

    /**
     * 设计音色
     */
    public VoiceDesignResponse designVoice(String speaker, String prompt) {
        // 调用 qwen-tts-vd API
        QwenVoiceDesignRequest request = new QwenVoiceDesignRequest()
                .setSpeaker(speaker)
                .setPrompt(prompt);
        
        QwenVoiceDesignResponse response = qwenTtsClient.post()
                .uri("/voice/design")
                .bodyValue(request)
                .retrieve()
                .bodyToMono(QwenVoiceDesignResponse.class)
                .block();
        
        log.info("Designed voice: {} for speaker: {}", response.getVoiceId(), speaker);
        
        return new VoiceDesignResponse(response.getVoiceId(), response.getAudioBase64());
    }

    /**
     * 删除音色
     */
    public void deleteVoice(String voiceId) {
        qwenTtsClient.delete()
                .uri("/voice/{voiceId}", voiceId)
                .retrieve()
                .bodyToMono(Void.class)
                .block();
        
        log.info("Deleted qwen voice: {}", voiceId);
    }
}

@Data
class QwenVoiceDesignRequest {
    private String speaker;
    private String prompt;
}

@Data
class QwenVoiceDesignResponse {
    private String voiceId;
    private String audioBase64;
}
```

```
    │
    ├── 无引用 → 调用百炼 API 清理
    │
    └── 有引用 → 保留百炼音色
```

### 12.2 清理逻辑

```java
/**
 * 删除角色时清理百炼音色
 */
public void cleanupBailianVoiceOnCharacterDelete(String userId, String excludeCharacterId) {
    // 1. 收集需要检查的百炼音色
    List<String> candidateVoiceIds = collectBailianVoiceIds(userId, excludeCharacterId);
    
    // 2. 检查每个音色是否还有引用
    for (String qwenVoiceId : candidateVoiceIds) {
        // 检查是否有其他角色引用
        boolean hasReference = characterMapper.existsByQwenVoiceId(qwenVoiceId, excludeCharacterId, userId);
        
        // 检查是否有独立的 GlobalVoice 记录
        boolean hasVoiceRecord = voiceMapper.existsByQwenVoiceId(qwenVoiceId, userId);
        
        if (!hasReference && !hasVoiceRecord) {
            // 无引用，清理百炼资源
            qwenTtsService.deleteVoice(qwenVoiceId);
            log.info("Cleaned up unreferenced qwen voice: {}", qwenVoiceId);
        }
    }
}
```

```java
package io.novel2video.agent.controller;

import io.novel2video.agent.dto.*;
import io.novel2video.agent.entity.*;
import io.novel2video.agent.service.*;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import java.util.List;

@Tag(name = "Voice API", description = "全局音色管理")
@RestController
@RequestMapping("/api/v1/assets/voices")
@RequiredArgsConstructor
public class VoiceController {

    private final VoiceService voiceService;
    private final UserService userService;

    private String getDefaultUserId() {
        return userService.getOrCreateDefaultUser().getUserId();
    }

    @Operation(summary = "获取音色列表")
    @GetMapping
    public ResponseEntity<ApiResponse<List<VoiceResponse>>> listVoices(
            @RequestParam(required = false) String folderId) {
        String userId = getDefaultUserId();
        List<GlobalVoice> voices = voiceService.listVoices(userId, folderId);
        List<VoiceResponse> response = voices.stream()
                .map(this::toVoiceResponse)
                .toList();
        return ResponseEntity.ok(ApiResponse.success(response));
    }

    @Operation(summary = "获取单个音色")
    @GetMapping("/{voiceId}")
    public ResponseEntity<ApiResponse<VoiceResponse>> getVoice(
            @PathVariable String voiceId) {
        GlobalVoice voice = voiceService.getVoice(voiceId);
        if (voice == null) {
            throw new RuntimeException("Voice not found: " + voiceId);
        }
        return ResponseEntity.ok(ApiResponse.success(toVoiceResponse(voice)));
    }

    @Operation(summary = "创建音色")
    @PostMapping
    public ResponseEntity<ApiResponse<VoiceResponse>> createVoice(
            @RequestBody @Valid CreateVoiceRequest request) {
        String userId = getDefaultUserId();
        GlobalVoice voice = voiceService.createVoice(userId, request);
        return ResponseEntity.ok(ApiResponse.success(toVoiceResponse(voice)));
    }

    @Operation(summary = "上传音频创建音色")
    @PostMapping("/upload")
    public ResponseEntity<ApiResponse<VoiceResponse>> uploadVoice(
            @RequestParam("file") MultipartFile file,
            @RequestParam("name") String name,
            @RequestParam(value = "folderId", required = false) String folderId,
            @RequestParam(value = "description", required = false) String description,
            @RequestParam(value = "gender", required = false) String gender,
            @RequestParam(value = "language", required = false) String language) {
        String userId = getDefaultUserId();
        GlobalVoice voice = voiceService.uploadVoice(userId, file, name, folderId, description, gender, language);
        return ResponseEntity.ok(ApiResponse.success(toVoiceResponse(voice)));
    }

    @Operation(summary = "更新音色")
    @PutMapping("/{voiceId}")
    public ResponseEntity<ApiResponse<VoiceResponse>> updateVoice(
            @PathVariable String voiceId,
            @RequestBody UpdateVoiceRequest request) {
        GlobalVoice voice = voiceService.updateVoice(voiceId, request);
        return ResponseEntity.ok(ApiResponse.success(toVoiceResponse(voice)));
    }

    @Operation(summary = "删除音色")
    @DeleteMapping("/{voiceId}")
    public ResponseEntity<ApiResponse<Void>> deleteVoice(
            @PathVariable String voiceId) {
        String userId = getDefaultUserId();
        voiceService.deleteVoice(voiceId, userId);
        return ResponseEntity.ok(ApiResponse.success(null));
    }

    @Operation(summary = "AI 设计音色")
    @PostMapping("/design")
    public ResponseEntity<ApiResponse<VoiceDesignResponse>> designVoice(
            @RequestBody @Valid VoiceDesignRequest request) {
        VoiceDesignResponse response = voiceService.designVoice(request);
        return ResponseEntity.ok(ApiResponse.success(response));
    }

    @Operation(summary = "音色选择器")
    @GetMapping("/picker")
    public ResponseEntity<ApiResponse<List<VoicePickerItem>>> getVoicePicker() {
        String userId = getDefaultUserId();
        List<GlobalVoice> voices = voiceService.getVoicePicker(userId);
        List<VoicePickerItem> items = voices.stream()
                .map(this::toVoicePickerItem)
                .toList();
        return ResponseEntity.ok(ApiResponse.success(items));
    }

    private VoiceResponse toVoiceResponse(GlobalVoice voice) {
        return new VoiceResponse(
                voice.getVoiceId(),
                voice.getFolderId(),
                voice.getName(),
                voice.getDescription(),
                voice.getQwenVoiceId(),
                voice.getVoiceType(),
                voice.getCustomVoiceUrl(),
                voice.getVoicePrompt(),
                voice.getGender(),
                voice.getLanguage(),
                voice.getCreatedAt()
        );
    }

    private VoicePickerItem toVoicePickerItem(GlobalVoice voice) {
        return new VoicePickerItem(
                voice.getVoiceId(),
                voice.getName(),
                voice.getDescription(),
                voice.getCustomVoiceUrl(),
                voice.getQwenVoiceId(),
                voice.getVoiceType(),
                voice.getGender(),
                voice.getLanguage()
        );
    }
}
```
