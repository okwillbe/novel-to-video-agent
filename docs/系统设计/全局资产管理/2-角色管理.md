# 全局角色管理设计文档

## 1. 概述

全局角色是用户级别的资产，存储于资产中心，可复用于多个项目。

| 特性 | 说明 |
|------|------|
| 归属级别 | 用户级，非项目级 |
| 主要功能 | 角色设定 + 多形象管理 + 音色绑定 |
| 复用机制 | 复制到项目（非引用） |
| AI 支持 | 从参考图生成、AI 设计形象、局部修改 |

## 6. 角色形象机制

### 6.1 多形象设计

```
角色: 张三
├── 形象 0 (appearanceIndex=0) - 默认形象
│   ├── changeReason: "初始形象"
│   ├── imageUrls: [图1, 图2, 图3, 图4]
│   └── selectedIndex: 2         ← 当前显示 图3
│
├── 形象 1 (appearanceIndex=1) - 便装
│   ├── changeReason: "便装"
│   ├── imageUrls: [图A, 图B]
│   └── selectedIndex: 0
│
└── 形象 2 (appearanceIndex=2) - 战甲
    └── changeReason: "战斗状态"
```

### 6.2 形象字段说明

| 字段 | 说明 |
|------|------|
| appearanceIndex | 形象序号，从 0 开始 |
| changeReason | 形象说明文字（如"便装"、"战甲"） |
| artStyle | 该形象的艺术风格 |
| description | 当前使用的描述词 |
| descriptions | 历史描述词列表（用于切换） |
| imageUrls | 生成的候选图片列表 |
| selectedIndex | 当前选中的图片索引 |

### 6.3 撤销机制

修改图片时：
1. 保存当前状态到 previous* 字段
2. 执行修改
3. 用户可调用 undo-image 撤销

```java
/**
 * 保存当前状态（支持撤销）
 */
private void savePreviousState(GlobalCharacterAppearance appearance) {
    appearance.setPreviousImageUrl(appearance.getImageUrl());
    appearance.setPreviousImageMediaId(appearance.getImageMediaId());
    appearance.setPreviousImageUrls(appearance.getImageUrls());
    appearance.setPreviousDescriptions(appearance.getDescriptions());
}

/**
 * 撤销图片修改
 */
@Transactional
public void undoImageChange(String appearanceId) {
    GlobalCharacterAppearance appearance = appearanceMapper.selectByAppearanceId(appearanceId);
    
    if (appearance.getPreviousImageUrls() == null) {
        throw new IllegalStateException("No previous state to undo");
    }
    
    // 恢复之前的状态
    appearance.setImageUrl(appearance.getPreviousImageUrl());
    appearance.setImageMediaId(appearance.getPreviousImageMediaId());
    appearance.setImageUrls(appearance.getPreviousImageUrls());
    appearance.setDescriptions(appearance.getPreviousDescriptions());
    
    // 清空 previous 字段
    appearance.setPreviousImageUrl(null);
    appearance.setPreviousImageMediaId(null);
    appearance.setPreviousImageUrls(null);
    appearance.setPreviousDescriptions(null);
    
    appearanceMapper.updateById(appearance);
}
```

## 7. Service 层实现

### 7.1 CharacterService

```java
package io.novel2video.agent.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import io.novel2video.agent.entity.*;
import io.novel2video.agent.mapper.*;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.*;

@Slf4j
@Service
@RequiredArgsConstructor
public class CharacterService {

    private final GlobalCharacterMapper characterMapper;
    private final GlobalCharacterAppearanceMapper appearanceMapper;
    private final TaskService taskService;

    /**
     * 获取角色列表
     */
    public List<GlobalCharacter> listCharacters(String userId, String folderId) {
        LambdaQueryWrapper<GlobalCharacter> wrapper = new LambdaQueryWrapper<GlobalCharacter>()
                .eq(GlobalCharacter::getUserId, userId);
        
        if ("null".equals(folderId)) {
            wrapper.isNull(GlobalCharacter::getFolderId);
        } else if (folderId != null) {
            wrapper.eq(GlobalCharacter::getFolderId, folderId);
        }
        
        wrapper.orderByDesc(GlobalCharacter::getCreatedAt);
        return characterMapper.selectList(wrapper);
    }

    /**
     * 获取单个角色（含形象）
     */
    public GlobalCharacter getCharacterWithAppearances(String characterId) {
        GlobalCharacter character = characterMapper.selectByCharacterId(characterId);
        if (character != null) {
            List<GlobalCharacterAppearance> appearances = listAppearances(characterId);
            // 设置到扩展字段或返回 DTO
        }
        return character;
    }

    /**
     * 创建角色
     */
    @Transactional
    public GlobalCharacter createCharacter(String userId, CreateCharacterRequest request) {
        // 1. 创建角色
        GlobalCharacter character = new GlobalCharacter()
                .setCharacterId(UUID.randomUUID().toString())
                .setUserId(userId)
                .setName(request.getName())
                .setFolderId(request.getFolderId())
                .setIntroduction(request.getDescription())
                .setProfileConfirmed(0);
        
        characterMapper.insert(character);
        log.info("Created character: {}", character.getCharacterId());
        
        // 2. 创建默认形象
        GlobalCharacterAppearance appearance = new GlobalCharacterAppearance()
                .setAppearanceId(UUID.randomUUID().toString())
                .setCharacterId(character.getCharacterId())
                .setAppearanceIndex(0)
                .setChangeReason("初始形象")
                .setArtStyle(request.getArtStyle())
                .setSelectedIndex(0);
        
        appearanceMapper.insert(appearance);
        
        // 3. 如果有参考图，触发 AI 生成任务
        if (Boolean.TRUE.equals(request.getGenerateFromReference()) 
                && request.getReferenceImageUrls() != null 
                && !request.getReferenceImageUrls().isEmpty()) {
            triggerImageGeneration(character.getCharacterId(), 0, request);
        }
        
        return character;
    }

    /**
     * 更新角色
     */
    public GlobalCharacter updateCharacter(String characterId, UpdateCharacterRequest request) {
        GlobalCharacter character = characterMapper.selectByCharacterId(characterId);
        if (character == null) {
            throw new RuntimeException("Character not found: " + characterId);
        }
        
        if (request.getName() != null) {
            character.setName(request.getName());
        }
        if (request.getAliases() != null) {
            character.setAliases(request.getAliases());
        }
        if (request.getProfileData() != null) {
            character.setProfileData(request.getProfileData());
        }
        if (request.getProfileConfirmed() != null) {
            character.setProfileConfirmed(request.getProfileConfirmed() ? 1 : 0);
        }
        if (request.getVoiceId() != null) {
            character.setVoiceId(request.getVoiceId());
        }
        if (request.getVoiceType() != null) {
            character.setVoiceType(request.getVoiceType());
        }
        if (request.getFolderId() != null) {
            character.setFolderId("null".equals(request.getFolderId()) ? null : request.getFolderId());
        }
        if (request.getGlobalVoiceId() != null) {
            character.setGlobalVoiceId(request.getGlobalVoiceId());
        }
        
        characterMapper.updateById(character);
        log.info("Updated character: {}", characterId);
        return character;
    }

    /**
     * 删除角色
     */
    @Transactional
    public void deleteCharacter(String characterId, String userId) {
        GlobalCharacter character = characterMapper.selectByCharacterId(characterId);
        if (character == null) {
            return;
        }
        
        // 1. 收集需要清理的音色ID
        List<String> candidateVoiceIds = new ArrayList<>();
        if (character.getVoiceId() != null && "qwen-designed".equals(character.getVoiceType())) {
            candidateVoiceIds.add(character.getVoiceId());
        }
        
        // 2. 删除所有形象
        LambdaQueryWrapper<GlobalCharacterAppearance> appearanceWrapper = 
                new LambdaQueryWrapper<GlobalCharacterAppearance>()
                        .eq(GlobalCharacterAppearance::getCharacterId, characterId);
        appearanceMapper.delete(appearanceWrapper);
        
        // 3. 删除角色
        characterMapper.deleteByCharacterId(characterId, userId);
        
        // 4. 清理无引用的百炼音色
        for (String voiceId : candidateVoiceIds) {
            // TODO: 检查是否有其他角色引用此音色，如果没有则清理
        }
        
        log.info("Deleted character: {}", characterId);
    }

    /**
     * 获取角色的所有形象
     */
    public List<GlobalCharacterAppearance> listAppearances(String characterId) {
        LambdaQueryWrapper<GlobalCharacterAppearance> wrapper = 
                new LambdaQueryWrapper<GlobalCharacterAppearance>()
                        .eq(GlobalCharacterAppearance::getCharacterId, characterId)
                        .orderByAsc(GlobalCharacterAppearance::getAppearanceIndex);
        return appearanceMapper.selectList(wrapper);
    }

    /**
     * 创建形象
     */
    public GlobalCharacterAppearance createAppearance(String characterId, CreateAppearanceRequest request) {
        // 获取当前最大索引
        Integer maxIndex = appearanceMapper.selectMaxIndexByCharacterId(characterId);
        int nextIndex = (maxIndex != null ? maxIndex : -1) + 1;
        
        GlobalCharacterAppearance appearance = new GlobalCharacterAppearance()
                .setAppearanceId(UUID.randomUUID().toString())
                .setCharacterId(characterId)
                .setAppearanceIndex(nextIndex)
                .setChangeReason(request.getChangeReason())
                .setArtStyle(request.getArtStyle())
                .setDescription(request.getDescription())
                .setSelectedIndex(0);
        
        appearanceMapper.insert(appearance);
        log.info("Created appearance: {} for character: {}", appearance.getAppearanceId(), characterId);
        return appearance;
    }

    /**
     * 选择候选图片
     */
    @Transactional
    public void selectImage(String appearanceId, Integer imageIndex) {
        GlobalCharacterAppearance appearance = appearanceMapper.selectByAppearanceId(appearanceId);
        
        if (appearance.getImageUrls() == null || imageIndex >= appearance.getImageUrls().size()) {
            throw new IllegalArgumentException("Invalid image index");
        }
        
        // 保存当前状态（支持撤销）
        savePreviousState(appearance);
        
        // 更新选中索引和当前图片
        appearance.setSelectedIndex(imageIndex);
        appearance.setImageUrl(appearance.getImageUrls().get(imageIndex));
        
        appearanceMapper.updateById(appearance);
        log.info("Selected image {} for appearance: {}", imageIndex, appearanceId);
    }

    /**
     * 触发图片生成任务
     */
    private String triggerImageGeneration(String characterId, Integer appearanceIndex, CreateCharacterRequest request) {
        Task task = taskService.createTask(Task.builder()
                .taskType(TaskType.ASSET_HUB_IMAGE)
                .targetType(TaskTargetType.GLOBAL_CHARACTER)
                .targetId(characterId)
                .inputData(Map.of(
                        "appearanceIndex", appearanceIndex,
                        "artStyle", request.getArtStyle(),
                        "referenceImageUrls", request.getReferenceImageUrls(),
                        "customDescription", request.getCustomDescription(),
                        "count", request.getCount() != null ? request.getCount() : 4
                ))
                .build());
        
        log.info("Triggered image generation task: {} for character: {}", task.getTaskId(), characterId);
        return task.getTaskId();
    }
}
```

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SSEService {

    private final SimpMessagingTemplate messagingTemplate;

    /**
     * 推送图片生成完成事件
     */
    public void pushImageGenerated(String userId, String characterId, Integer appearanceIndex, List<String> imageUrls) {
        SSEEvent event = SSEEvent.builder()
                .type("IMAGE_GENERATED")
                .data(Map.of(
                        "characterId", characterId,
                        "appearanceIndex", appearanceIndex,
                        "imageUrls", imageUrls
                ))
                .build();
        
        pushToUser(userId, event);
        log.info("Pushed IMAGE_GENERATED event to user: {}", userId);
    }

    /**
     * 推送任务进度
     */
    public void pushTaskProgress(String userId, String taskId, Integer progress, String status) {
        SSEEvent event = SSEEvent.builder()
                .type("TASK_PROGRESS")
                .data(Map.of(
                        "taskId", taskId,
                        "progress", progress,
                        "status", status
                ))
                .build();
        
        pushToUser(userId, event);
    }

    private void pushToUser(String userId, SSEEvent event) {
        messagingTemplate.convertAndSendToUser(userId, "/queue/events", event);
    }
}
```
